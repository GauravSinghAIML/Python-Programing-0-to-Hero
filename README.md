# Python Operations
# 1. Arithmetical Operations
    x = 14
    y = 3
    #Addition - "+"
    print('x + y =',x+y) -----14 + 3 = 17

    # Subtraction - "-"
    print('x - y =',x-y)  ----- 14 - 3 = 11

    # Multiplication - "*"
    print('x * y =',x*y)   -----14* 3 = 42

    # Division - "/" 
    print('x / y =',x/y)  ----- 14/3 = 4.66666

    # Floor Division - "//"
    print('x // y =',x//y) ---------14//3 = 4

    # Modulus or remainder - "%"
    print('x % y =',x%y)    ---------14%3 = 2

    # Exponent  - "**"
    print('x ** y =',x**y  ---------14**3 = 2744
